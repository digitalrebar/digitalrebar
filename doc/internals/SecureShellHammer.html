def run_on(cmd)
  Rails.logger.debug("Node: #{node.name}: Running #{cmd}")
  out,err = '',''
  status = Open4::popen4ext(true,cmd) do |pid,stdin,stdout,stderr|
    stdin.close
    out << stdout.read
    err << stderr.read
    stdout.close
    stderr.close
  end
  [out,err,status]
end
