def template
  # this is a workable solution for now, we use the admin node to determine domain (except when non-exists!)
  domain = Node.admin.first.name.split(".",2)[1] rescue I18n.t('not_set')
  {"crowbar" => {     "dns" => {
                      "domain" => domain,
                      "contact" => "support@localhost.localdomain",
                      "forwarders" =>  [],
                      "static" => {},
                      "ttl" => "1h",
                      "slave_refresh" => "1d",
                      "slave_retry" => "2h",

                      "negative_cache" => 300}}}
end
